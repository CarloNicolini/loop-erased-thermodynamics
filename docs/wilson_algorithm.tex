\begin{algorithm}[htb]
    \SetAlgoLined
    \DontPrintSemicolon
    \KwInput{$G = (V,E,w)$ a weighted undirected graph; optional root weight $q \ge 0$.}
    \KwOutput{$F$ a random spanning forest; $\mathcal{R}$ the set of roots.}
    
    \BlankLine
    \If{$q > 0$}{
      Add auxiliary root $r$ and edges $(v,r)$ of weight $q$ for all $v \in V$\;
    }
    Choose root $r$ (auxiliary if $q>0$, otherwise arbitrary)\;
    Initialize forest $F \leftarrow \emptyset$, root set $\mathcal{R} \leftarrow \{r\}$\;
    Mark all vertices as unvisited\;
    
    \BlankLine
    \For{each unvisited vertex $v$ in random order}{
      Perform a random walk from $v$ until it reaches a visited node,
      choosing neighbors with probability $\propto w(u,w)$\;
      Erase loops from the walk to obtain a simple path $P$\;
      Add $P$ to $F$ and mark its vertices as visited\;
    }
    
    \BlankLine
    \If{$q > 0$}{
      Let $\mathcal{R}$ be all vertices directly linked to the auxiliary root\;
    }
    \Return{$(F, \mathcal{R})$}
    \caption{Random Rooted Forest Sampling (Wilson $q$-algorithm)}
    \label{alg:wilson}
    \end{algorithm}
